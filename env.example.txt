# Copy this to .env for local dev. On Railway, set these as project Variables instead.
#
# NOTE: the /admin.html dashboard has no login — anyone with the URL can view,
# create, and delete vehicles, and its endpoints can trigger Shopify product
# searches and file uploads through this app's credentials. Treat the Railway
# URL as a secret in itself: don't publish it, don't link to it from the
# public storefront, and only share it with people who should have edit
# access. This is fine for a single-person internal tool; if that stops being
# true, a login is worth re-adding.

PORT=3000

# Your store's .myshopify.com domain (not your custom domain), e.g. 4x4factorysa.myshopify.com
SHOPIFY_STORE_DOMAIN=your-store.myshopify.com

# Same Client ID / Secret pattern used for Frikkie (Client Credentials Grant,
# new dev-dashboard custom app model). Needs read_products + write_files scopes.
SHOPIFY_CLIENT_ID=
SHOPIFY_CLIENT_SECRET=

# Admin API version to target — bump this periodically.
SHOPIFY_API_VERSION=2025-01

# Your live storefront domain, used to build "View product" links,
# e.g. https://4x4factory.co.za  (no trailing slash)
STOREFRONT_URL=https://your-store-domain.co.za

# Where the SQLite file lives. On Railway this MUST point into a mounted
# persistent volume (e.g. /data/hotspots.db) or every redeploy wipes your
# vehicles/hotspots — same gotcha you already hit with Frikkie.
DATABASE_PATH=./data/hotspots.db

# --- Quote request emails ---
# Any SMTP account works: Gmail with an app password, your hosting
# provider's email, or a transactional service like SendGrid/Postmark (they
# all support plain SMTP too). Leave these blank to skip email notifications
# — draft orders will still be created in Shopify either way, you'd just
# have to check for them manually instead of getting an email.
SMTP_HOST=
SMTP_PORT=587
SMTP_USER=
SMTP_PASS=
# Optional — defaults to SMTP_USER if blank.
SMTP_FROM=
# Where quote request notifications get sent.
NOTIFY_EMAIL=

# --- Branding ---
# Shown in the admin dashboard header, and on the "Save my build" page
# customers print/save. Point this at a logo image URL (e.g. one already on
# your Shopify CDN). Leave blank to skip showing a logo anywhere.
STORE_LOGO_URL=
# Optional footer line on the "Save my build" page, under the total —
# e.g. a phone number / website / address.
STORE_CONTACT=
